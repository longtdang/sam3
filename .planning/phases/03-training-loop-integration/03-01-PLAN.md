```python
class ColorJitter:
    """API-compatible ColorJitter wrapper for use inside ComposeAPI transform pipelines.

    Wraps torchvision.transforms.ColorJitter. Applies the same jitter to every image
    in the datapoint (PIL stage — place before ToTensorAPI).

    Per D-03-05: brightness=0.2, contrast=0.2, saturation=0.2, hue=0.0 for industrial data.
    """

    def __init__(self, *args, **kwargs):
        self.jitter = T.ColorJitter(*args, **kwargs)

    def __call__(self, datapoint, **kwargs):
        for img in datapoint.images:
            img.data = self.jitter(img.data)
        return datapoint


class GaussianBlur:
    """API-compatible GaussianBlur wrapper for use inside ComposeAPI transform pipelines.

    Wraps torchvision.transforms.GaussianBlur. Applies the same blur to every image
    in the datapoint (PIL stage — place before ToTensorAPI).

    Per D-03-05: kernel_size=3, sigma=(0.1, 2.0) to simulate sensor noise.
    """

    def __init__(self, *args, **kwargs):
        self.blur = T.GaussianBlur(*args, **kwargs)

    def __call__(self, datapoint, **kwargs):
        for img in datapoint.images:
            img.data = self.blur(img.data)
        return datapoint


class RandomErasingAPI:
    """API-compatible RandomErasing wrapper for use inside ComposeAPI transform pipelines.

    Distinct from the existing RandomErasing(img, target) class above — this class uses
    the (datapoint, **kwargs) API signature. Place AFTER ToTensorAPI (tensor input required)
    and BEFORE NormalizeAPI.

    Per D-03-05: p=0.2, scale=(0.02, 0.1) for small occlusion patches.
    """

    def __init__(self, *args, **kwargs):
        self.eraser = T.RandomErasing(*args, **kwargs)

    def __call__(self, datapoint, **kwargs):
        for img in datapoint.images:
            img.data = self.eraser(img.data)
        return datapoint
```

Do NOT modify any existing classes. Do NOT change imports. Append only.
