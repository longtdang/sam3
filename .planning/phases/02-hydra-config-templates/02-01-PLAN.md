# ============================================================================

  # --- Segmentation toggle (CRITICAL: SAM3 defaults to detection-only; this enables mask training) ---

  val_norm_mean: [0.5, 0.5, 0.5]
  val_norm_std: [0.5, 0.5, 0.5]

  target_epoch_size: 500  # documentation-only — not consumed by Trainer code

  context_length: 2       # text query token count (do not change)

  lr_transformer: 8e-5     # D-12: transformer / cross-attention / class head LR
  lr_vision_backbone: 2.5e-6  # D-12: ViT trunk LR (low → near-frozen in decoder-only)

  val_batch_size: 1

            size: ${scratch.resolution}
          square: true
          consistent_transform: ${scratch.consistent_transform}
        - _target_: sam3.train.transforms.basic_for_api.PadToSizeAPI
          size: ${scratch.resolution}  # pad to 1008×1008 square
          consistent_transform: ${scratch.consistent_transform}
        - _target_: sam3.train.transforms.basic_for_api.ToTensorAPI
        - _target_: sam3.train.transforms.filter_query_transforms.FlexibleFilterFindGetQueries
          query_filter:
            _target_: sam3.train.transforms.filter_query_transforms.FilterEmptyTargets  # drop images with no valid boxes post-resize
        - _target_: sam3.train.transforms.basic_for_api.NormalizeAPI
          mean: ${scratch.train_norm_mean}  # CFG-06: [0.5, 0.5, 0.5]
          std: ${scratch.train_norm_std}
        - _target_: sam3.train.transforms.filter_query_transforms.FlexibleFilterFindGetQueries

            _target_: sam3.train.transforms.filter_query_transforms.FilterEmptyTargets
    - _target_: sam3.train.transforms.filter_query_transforms.FlexibleFilterFindGetQueries

        _target_: sam3.train.transforms.filter_query_transforms.FilterFindQueriesWithTooManyOut

          sizes: ${scratch.resolution}  # deterministic single-scale resize
          max_size:
            _target_: sam3.train.transforms.basic.get_random_resize_max_size
            size: ${scratch.resolution}
          square: true
          consistent_transform: False
        - _target_: sam3.train.transforms.basic_for_api.ToTensorAPI
        - _target_: sam3.train.transforms.basic_for_api.NormalizeAPI
          mean: ${scratch.val_norm_mean}  # CFG-06: [0.5, 0.5, 0.5]

      _target_: sam3.train.matcher.BinaryOneToManyMatcher
      alpha: 0.3
      threshold: 0.4

    loss_fns_find:

        weight_dict:
          loss_bbox: 5.0
          loss_giou: 2.0
      - _target_: sam3.train.loss.loss_fns.IABCEMdetr  # classification + presence focal loss
        weak_loss: False
        weight_dict:
          loss_ce: 20.0
          presence_loss: 20.0

        alpha: 0.25
        gamma: 2
        use_presence: True
        pos_focal: false

        compute_aux: false
    loss_fn_semantic_seg: null  # optional semantic-seg head; null = disabled for image-only fine-tune
